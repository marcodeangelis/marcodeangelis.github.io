					<section>
						<h3>Goals</h3>
						<ul>
							<li style="margin: 35px;"> Objectively measure the conformance of predictions with empirical data
							<li style="margin: 35px;">  Use this measure to characterize the reliability of other predictions
						</ul>
					</section>
					<section>
					<h3>Validation metric</h3>
						<ul>
							<li style="margin: 35px;"> A measure of the mismatch between the observed data and the model’s predictions
								<ul style="color:green; font-size: 30px;">
									<li> 	Low value means a good match
									<li>	High value means they disagree
								</ul>
								<li style="margin: 35px;">  Distance between prediction and data
						</ul>
					</section>
					<section>
					<h3>Desirable properties of such metric</h3>
						<ul>
							<li style="margin: 30px;"> Expressed in physical units
							<li style="margin: 30px;"> Generalizes deterministic comparisons
							<li style="margin: 30px;"> Reflects full distribution
							<li style="margin: 30px;"> Not too sensitive to long tails
							<li style="margin: 30px;"> Mathematical metric
							<li style="margin: 30px;"> Unbounded (you can be really off)

						</ul>
					</section>
					<section>
						<h3>Area metric</h3>
						<ul style="font-size: 32px;">
							<li style="margin: 10px;"> "Distance" between two distributions
							<li style="margin: 10px;"> Physical units
							<li style="margin: 10px;"> Smallest mean absolute difference of deviates
						</ul>
						<p style="position: absolute; right:0px; top:250px; color:grey"> \[ min\ E \left| X_F - Y_n \right| \]</p>
